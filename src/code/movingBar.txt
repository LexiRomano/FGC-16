# 0x8000 0x8F00
Var blipChar
LDA 219
STA blipChar

Var notBlipChar
LDA 176
STA notBlipChar

Var length
LDA 100
STA length

Var blipLoc
Var pnt

:start
LDA 0xFF
STA @0x6003

:loop
LDA blipLoc
SUB pnt
JZR equals

LDA notBlipChar

:draw
STA @0x6000

LDA pnt
INA
STA pnt
SUB length
JZR doneLine
JMP loop

:doneLine
LDA 0
STA pnt
LDA blipLoc
INA
STA blipLoc
SUB length
JZR resetBlipLoc
JMP start

:equals
LDA blipChar
JMP draw

:resetBlipLoc
LDA 0
STA blipLoc
JMP start
